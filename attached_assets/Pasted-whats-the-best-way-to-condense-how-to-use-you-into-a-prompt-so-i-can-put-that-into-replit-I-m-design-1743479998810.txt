2. Add Search with Autocomplete
You can implement search with autocomplete using either the Mapbox Geocoding API or the Search Box API. Since you need autocomplete and want to check if locations are within a specific area, the Search Box API is more appropriate as it includes POIs and better autocomplete features Unlock faster, smarter search with Mapbox Search Box API.
Here's how to implement it:

mapboxgl.accessToken = 'YOUR_MAPBOX_ACCESS_TOKEN';

// Initialize the map centered on Southern California
const map = new mapboxgl.Map({
  container: 'map',
  style: 'mapbox://styles/mapbox/streets-v12',
  center: [-118.2437, 34.0522], // Los Angeles coordinates
  zoom: 8
});

// Define your service area boundary (simplified example - you would define your actual service area)
const serviceAreaBoundary = {
  type: 'Polygon',
  coordinates: [[
    // Define coordinates for Southern California boundary
    // This is a simplified example - you would need actual boundary coordinates
    [-119.5, 33.5],
    [-117.0, 33.5],
    [-117.0, 34.5],
    [-119.5, 34.5],
    [-119.5, 33.5]
  ]]
};

// Add the geocoder (search box with autocomplete)
const geocoder = new MapboxGeocoder({
  accessToken: mapboxgl.accessToken,
  mapboxgl: mapboxgl,
  proximity: {
    longitude: -118.2437,
    latitude: 34.0522
  }, // Bias results toward Southern California
  countries: 'us',
  bbox: [-124.409591, 32.534156, -114.131211, 37.0], // Bounding box for California
  placeholder: 'Search for a location'
});

// Add the geocoder to the map
map.addControl(geocoder);

// When a result is selected, check if it's in your service area
geocoder.on('result', function(e) {
  const resultCoordinates = e.result.geometry.coordinates;
  const point = turf.point(resultCoordinates);
  
  // Check if the point is within the service area
  const isInServiceArea = turf.booleanPointInPolygon(point, turf.polygon(serviceAreaBoundary.coordinates));
  
  const resultDiv = document.getElementById('result');
  if (isInServiceArea) {
    resultDiv.innerHTML = '<h3>Good news!</h3><p>We serve this location.</p>';
    resultDiv.style.backgroundColor = '#d4edda';
  } else {
    resultDiv.innerHTML = '<h3>Sorry!</h3><p>We don\'t currently serve this location.</p>';
    resultDiv.style.backgroundColor = '#f8d7da';
  }
  
  // Add a marker at the selected location
  new mapboxgl.Marker()
    .setLngLat(resultCoordinates)
    .addTo(map);
});

// Add Turf.js for spatial analysis
map.on('load', function() {
  // Add your service area as a layer on the map
  map.addSource('service-area', {
    type: 'geojson',
    data: {
      type: 'Feature',
      geometry: serviceAreaBoundary
    }
  });
  
  map.addLayer({
    id: 'service-area-fill',
    type: 'fill',
    source: 'service-area',
    paint: {
      'fill-color': '#088',
      'fill-opacity': 0.3
    }
  });
  
  map.addLayer({
    id: 'service-area-outline',
    type: 'line',
    source: 'service-area',
    paint: {
      'line-color': '#088',
      'line-width': 2
    }
  });
});
